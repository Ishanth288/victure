      .from("patients")

        name: cleanPatientName,

        console.log("Duplicate phone number detected, fetching existing patient...");

          .from("patients")

        if (duplicatePatient) {
          patientData = duplicatePatient;
          break;
        }
      }
      throw createPatientError;

    patientData = newPatient;
